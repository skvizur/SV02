create table movies (
title varchar(250),
year integer,
movieid integer,
primary key (movieid));

create table username (
gender varchar(250),
age varchar(250),
occupation integer,
userid integer,
primary key (userid)
);

create table rating (
movie_id integer references movies(movieid),
rating integer,
user_id integer references username(userid),
primary key (movie_id, user_id)
);

create table genre_id (
movie_id integer references movies(movieid),
genreid integer,
primary key (movie_id)
);
