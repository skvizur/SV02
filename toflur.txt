create table movies (
title varchar(250),
genres varchar(250),
movieid integer,
primary key (movieid));

create table username (
gender varchar(250),
age varchar(250),
occupation integer,
zipcode varchar(250),
userid integer,
primary key (userid)
);

create table rating (
movie_id integer references movies(movieid),
rating integer,
"timestamp" integer,
user_id integer references username(userid),
primary key (movie_id, user_id)
);
